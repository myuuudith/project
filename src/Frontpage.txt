  setLike(like-1)  
}else{
  setLikeActive(true)
  setLike(like+1)
  if(dislikeActive){

    setLike(like+1)
    setDislike(dislike-1)
  }
}
}

    setLike(like+1)

      setLikeActive(false)
      setDislike(dislike+1)
      setLike(like-1)
    }
  }
}
